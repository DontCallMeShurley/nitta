   39 {- | The constant, which restricts the maximum number of synthesis steps. Avoids
   40 the endless synthesis process.
   41 -}
   42 stepLimit = 750 :: Int
   43 
   44 noSynthesis :: () -> SynthesisMethod tag v x t
   45 noSynthesis () tree = do
   46     infoM "NITTA.Synthesis" "noSynthesis"
   47     return tree
   48 
   49 -- | The most complex synthesis method, which embedded all another. That all.
   50 stateOfTheArtSynthesisIO :: (VarValTime v x t, UnitTag tag) => () -> SynthesisMethod tag v x t
   51 stateOfTheArtSynthesisIO () tree = do
   52     infoM "NITTA.Synthesis" $ "stateOfTheArtSynthesisIO: " <> show (sID tree)
   53     l1 <- simpleSynthesisIO tree
   54     l2 <- smartBindSynthesisIO tree
   55     l3 <- bestThreadIO stepLimit tree
   56     l4 <- bestThreadIO stepLimit =<< allBindsAndRefsIO tree
   57     l5 <- obviousGroupBindsIO tree >>= tryAllGroupBindsByIO (bestThreadIO stepLimit)
   58     return $ bestLeaf tree [l1, l2, l3, l4, l5]
   59 
   60 -- | Schedule process by simple synthesis.
   61 simpleSynthesisIO :: (VarValTime v x t, UnitTag tag) => SynthesisMethod tag v x t
   62 simpleSynthesisIO root = do
   63     infoM "NITTA.Synthesis" $ "simpleSynthesisIO: " <> show (sID root)
   64     lastObviousNode <- obviousBindThreadIO root
   65     allBestThreadIO 1 lastObviousNode
   66 
   67 smartBindSynthesisIO :: (VarValTime v x t, UnitTag tag) => SynthesisMethod tag v x t
   68 smartBindSynthesisIO tree = do
   69     infoM "NITTA.Synthesis" $ "smartBindSynthesisIO: " <> show (sID tree)
   70     tree' <- smartBindThreadIO tree
   71     allBestThreadIO 1 tree'
   72 
   73 bestThreadIO :: (VarValTime v x t, UnitTag tag) => Int -> SynthesisMethod tag v x t
   74 bestThreadIO 0 node = return $ trace "bestThreadIO reach step limit!" node
   75 bestThreadIO limit tree = do
   76     subForest <- positiveSubForestIO tree
   77     case subForest of
   78         [] -> return tree
   79         _ -> bestThreadIO (limit - 1) $ maximumOn (defScore . sDecision) subForest
   80 
   81 bestStepIO :: (VarValTime v x t, UnitTag tag) => SynthesisMethod tag v x t
   82 bestStepIO tree = do
   83     subForest <- positiveSubForestIO tree
   84     case subForest of
   85         [] -> error "all step is over"
   86         _ -> return $ maximumOn (defScore . sDecision) subForest
   87 
   88 obviousGroupBindsIO :: (VarValTime v x t, UnitTag tag) => SynthesisMethod tag v x t
   89 obviousGroupBindsIO tree = do
   90     binds <- selectSubForestIO isObviousMultiBind tree
   91     maybe (return tree) obviousGroupBindsIO $ bestDecision binds
   92 
   93 tryAllGroupBindsByIO :: (VarValTime v x t, UnitTag tag) => SynthesisMethod tag v x t -> SynthesisMethod tag v x t
   94 tryAllGroupBindsByIO method tree = do
   95     bindSubForest <- selectSubForestIO isMultiBind tree
   96     leafs <- mapM method bindSubForest
   97     return $ bestLeaf tree leafs
   98 
   99 obviousBindThreadIO :: (VarValTime v x t, UnitTag tag) => SynthesisMethod tag v x t
  100 obviousBindThreadIO tree = do
  101     subForest <- positiveSubForestIO tree
  102     maybe (return tree) obviousBindThreadIO $
  103         L.find
  104             ( ( \case
  105                     Just SingleBindMetrics{pPossibleDeadlock = True} -> False
  106                     Just SingleBindMetrics{pAlternative = 1} -> True
  107                     _ -> False
  108               )
  109                 . cast
  110                 . sDecision
  111             )
  112             subForest
  113 
  114 allBindsAndRefsIO :: (VarValTime v x t, UnitTag tag) => SynthesisMethod tag v x t
  115 allBindsAndRefsIO tree = do
  116     subForest <-
  117         filter ((\d -> isSingleBind d || isRefactor d) . sDecision)
  118             <$> positiveSubForestIO tree
  119     case subForest of
  120         [] -> return tree
  121         _ -> allBindsAndRefsIO $ maximumOn (defScore . sDecision) subForest
  122 
  123 refactorThreadIO :: (VarValTime v x t, UnitTag tag) => SynthesisMethod tag v x t
  124 refactorThreadIO tree = do
  125     subForest <- positiveSubForestIO tree
  126     maybe (return tree) refactorThreadIO $
  127         L.find (isRefactor . sDecision) subForest
  128 
  129 smartBindThreadIO :: (VarValTime v x t, UnitTag tag) => SynthesisMethod tag v x t
  130 smartBindThreadIO tree = do
  131     subForest <-
  132         filter ((\d -> isSingleBind d || isRefactor d) . sDecision)
  133             <$> (positiveSubForestIO =<< refactorThreadIO tree)
  134     case subForest of
  135         [] -> return tree
  136         _ -> smartBindThreadIO $ maximumOn (defScore . sDecision) subForest
  137 
  138 allBestThreadIO :: (VarValTime v x t, UnitTag tag) => Int -> SynthesisMethod tag v x t
  139 allBestThreadIO (0 :: Int) tree = bestThreadIO stepLimit tree
  140 allBestThreadIO n tree = do
  141     subForest <- positiveSubForestIO tree
  142     leafs <- mapM (allBestThreadIO (n - 1)) subForest
  143     return $ bestLeaf tree leafs
  144 
  145 bestLeaf :: (VarValTime v x t, UnitTag tag) => DefTree tag v x t -> [DefTree tag v x t] -> DefTree tag v x t
  146 bestLeaf tree leafs =
  147     let successLeafs = filter (\node -> isComplete node && isLeaf node) leafs
  148      in case successLeafs of
  149             [] -> headDef tree leafs
  150             _ : _ ->
  151                 minimumOn
  152                     (\Tree{sState = SynthesisState{sTarget}} -> (processDuration sTarget, puSize sTarget))
  153                     successLeafs
  154 
  155 -- * Helpers
  156 
  157 selectSubForestIO ::
  158     ( UnitTag tag
  159     , VarValTime v x t
  160     , m ~ TargetSystem (BusNetwork tag v x t) tag v x t
  161     , ctx ~ SynthesisState m tag v x t
  162     ) =>
  163     (SynthesisDecision ctx m -> Bool) ->
  164     DefTree tag v x t ->
  165     IO [DefTree tag v x t]
  166 selectSubForestIO p tree = filter (p . sDecision) <$> positiveSubForestIO tree
  167 
  168 bestDecision :: [DefTree tag v x t] -> Maybe (DefTree tag v x t)
  169 bestDecision [] = Nothing
  170 bestDecision xs = Just $ maximumOn (defScore . sDecision) xs
