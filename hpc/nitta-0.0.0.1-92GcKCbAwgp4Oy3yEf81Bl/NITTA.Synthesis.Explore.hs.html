   41 -- |Make synthesis tree
   42 synthesisTreeRootIO = atomically . rootSynthesisTreeSTM
   43 
   44 rootSynthesisTreeSTM model = do
   45     sSubForestVar <- newEmptyTMVar
   46     return
   47         Tree
   48             { sID = def
   49             , sState = nodeCtx Nothing model
   50             , sDecision = Root
   51             , sSubForestVar
   52             }
   53 
   54 -- |Get specific by @nId@ node from a synthesis tree.
   55 getTreeIO tree (Sid []) = return tree
   56 getTreeIO tree (Sid (i : is)) = do
   57     subForest <- subForestIO tree
   58     unless (i < length subForest) $ error "getTreeIO - wrong Sid"
   59     getTreeIO (subForest !! i) (Sid is)
   60 
   61 -- |Get list of all nodes from root to selected.
   62 getTreePathIO _ (Sid []) = return []
   63 getTreePathIO tree (Sid (i : is)) = do
   64     h <- getTreeIO tree $ Sid [i]
   65     t <- getTreePathIO h $ Sid is
   66     return $ h : t

   68 {- |Get all available edges for the node. Edges calculated only for the first
   69 call.
   70 -}
   71 subForestIO
   72     tree@Tree
   73         { sSubForestVar
   74         , sID
   75         , sDecision
   76         } = do
   77         (firstTime, subForest) <-
   78             atomically $
   79                 tryReadTMVar sSubForestVar >>= \case
   80                     Just subForest -> return (False, subForest)
   81                     Nothing -> do
   82                         subForest <- exploreSubForestVar tree
   83                         putTMVar sSubForestVar subForest
   84                         return (True, subForest)
   85         when firstTime $ do
   86             debugM "NITTA.Synthesis" $
   87                 "explore: "
   88                     <> show sID
   89                     <> " score: "
   90                     <> ( case sDecision of
   91                             SynthesisDecision{score} -> show score
   92                             _ -> "-"
   93                        )
   94                     <> " decision: "
   95                     <> ( case sDecision of
   96                             SynthesisDecision{decision} -> show decision
   97                             _ -> "-"
   98                        )
   99 
  100         return subForest
  101 
  102 {- |For synthesis method is more usefull, because throw away all useless trees in
  103 subForest (objective function value less than zero).
  104 -}
  105 positiveSubForestIO tree = filter ((> 0) . score . sDecision) <$> subForestIO tree

  107 isLeaf
  108     Tree
  109         { sState =
  110             SynthesisState
  111                 { sAllocationOptions = []
  112                 , sBindOptions = []
  113                 , sDataflowOptions = []
  114                 , sBreakLoopOptions = []
  115                 , sResolveDeadlockOptions = []
  116                 , sOptimizeAccumOptions = []
  117                 , sConstantFoldingOptions = []
  118                 }
  119         } = True
  120 isLeaf _ = False
  121 
  122 isComplete = isSynthesisComplete . sTarget . sState

  125 exploreSubForestVar parent@Tree{sID, sState} =
  126     let edges =
  127             concat
  128                 ( map (decisionAndContext parent) (sAllocationOptions sState)
  129                     ++ map (decisionAndContext parent) (sBindOptions sState)
  130                     ++ map (decisionAndContext parent) (sDataflowOptions sState)
  131                     ++ map (decisionAndContext parent) (sBreakLoopOptions sState)
  132                     ++ map (decisionAndContext parent) (sResolveDeadlockOptions sState)
  133                     ++ map (decisionAndContext parent) (sOptimizeAccumOptions sState)
  134                     ++ map (decisionAndContext parent) (sConstantFoldingOptions sState)
  135                 )
  136      in forM (zip [0 ..] edges) $ \(i, (desc, ctx')) -> do
  137             sSubForestVar <- newEmptyTMVar
  138             return
  139                 Tree
  140                     { sID = sID <> Sid [i]
  141                     , sState = ctx'
  142                     , sDecision = desc
  143                     , sSubForestVar
  144                     }
  145 
  146 decisionAndContext parent@Tree{sState = ctx} o =
  147     [ (SynthesisDecision o d p e, nodeCtx (Just parent) model)
  148     | (d, model) <- decisions ctx o
  149     , let p = parameters ctx o d
  150           e = estimate ctx o d p
  151     ]
  152 
  153 nodeCtx parent nModel =
  154     let sBindOptions = bindOptions nModel
  155         sDataflowOptions = dataflowOptions nModel
  156         fs = functions $ mDataFlowGraph nModel
  157         processWaves = buildProcessWaves [] fs
  158      in SynthesisState
  159             { sTarget = nModel
  160             , sParent = parent
  161             , sAllocationOptions = allocationOptions nModel
  162             , sBindOptions
  163             , sDataflowOptions
  164             , sResolveDeadlockOptions = resolveDeadlockOptions nModel
  165             , sBreakLoopOptions = breakLoopOptions nModel
  166             , sConstantFoldingOptions = constantFoldingOptions nModel
  167             , sOptimizeAccumOptions = optimizeAccumOptions nModel
  168             , bindingAlternative =
  169                 foldl
  170                     (\st (Bind f tag) -> M.alter (return . maybe [tag] (tag :)) f st)
  171                     M.empty
  172                     sBindOptions
  173             , possibleDeadlockBinds =
  174                 S.fromList
  175                     [ f
  176                     | (Bind f tag) <- sBindOptions
  177                     , Lock{lockBy} <- locks f
  178                     , lockBy `S.member` unionsMap variables (bindedFunctions tag $ mUnit nModel)
  179                     ]
  180             , bindWaves = estimateVarWaves (S.elems (variables (mUnit nModel) S.\\ unionsMap variables sBindOptions)) fs
  181             , processWaves
  182             , numberOfProcessWaves = length processWaves
  183             , numberOfDataflowOptions = length sDataflowOptions
  184             , transferableVars =
  185                 S.unions
  186                     [ variables ep
  187                     | (DataflowSt _ targets) <- sDataflowOptions
  188                     , (_, ep) <- targets
  189                     ]
  190             }
