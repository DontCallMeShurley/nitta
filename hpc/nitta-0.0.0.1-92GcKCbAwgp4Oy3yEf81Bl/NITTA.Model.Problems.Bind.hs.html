   17 import Data.String.ToString
   18 import GHC.Generics
   19 import NITTA.Intermediate.Types
   20 
   21 data Bind tag v x
   22     = Bind (F v x) tag
   23     deriving (Generic, Eq)
   24 
   25 instance (ToString tag) => Show (Bind tag v x) where
   26     show (Bind f tag) = "Bind " <> show f <> " " <> toString tag
   27 
   28 class BindProblem u tag v x | u -> tag v x where
   29     bindOptions :: u -> [Bind tag v x]
   30     bindDecision :: u -> Bind tag v x -> u
   31 
   32 instance (Var v) => Variables (Bind tab v x) v where
   33     variables (Bind f _tag) = variables f
