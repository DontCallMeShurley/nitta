   14 Synthesis can be represented as a graph (tree), where each 'Node' describes the
   15 target system 'ModelState' and each 'Edge' synthesis decision.
   16 
   17 A synthesis graph is very large and calculating and storing it in memory is very
   18 bad idea. Also, working with synthesis graph usually making from the specific
   19 node, not from the root. As a result, synthesis graph design as a explicit lazy
   20 mutable structure implemented by 'TVar'.
   21 
   22 From this point of view, the synthesis process is a finding of the best tree
   23 leaf (lowest process duration for finished synthesis), and the best synthesis
   24 method - a method which directly walks over the tree to the best leaf without
   25 wrong steps.
   26 -}

   61 -- | Default synthesis tree type.
   62 type DefTree tag v x t =
   63     Tree (TargetSystem (BusNetwork tag v x t) tag v x t) tag v x t
   64 
   65 {- | The synthesis method is a function, which manipulates a synthesis tree. It
   66 receives a node and explores it deeply by IO.
   67 -}
   68 type SynthesisMethod tag v x t = DefTree tag v x t -> IO (DefTree tag v x t)

   70 {- | Synthesis node ID. ID is a relative path, encoded as a sequence of an option
   71 index.
   72 -}
   73 newtype Sid = Sid [Int]
   74 
   75 -- | Sid separator for @Show Sid@ and @Read Sid@.
   76 sidSep = '-'
   77 
   78 instance Show Sid where
   79     show (Sid []) = [sidSep]
   80     show (Sid is) = show' is
   81         where
   82             show' [] = ""
   83             show' (x : xs) = sidSep : show x ++ show' xs
   84 
   85 instance Read Sid where
   86     readsPrec _ [x] | x == sidSep = [(Sid [], "")]
   87     readsPrec d (x : xs)
   88         | x == sidSep
   89         , let is = map (readsPrec d) $ splitOn [sidSep] xs
   90         , not $ any null is =
   91             [(Sid $ map fst $ concat is, "")]
   92     readsPrec _ _ = []
   93 
   94 instance Default Sid where
   95     def = Sid []
   96 
   97 instance Semigroup Sid where
   98     (Sid a) <> (Sid b) = Sid (a <> b)
   99 
  100 instance Monoid Sid where
  101     mempty = Sid []
  102     mappend = (<>)
  103 
  104 instance ToJSON Sid where
  105     toJSON sid = toJSON $ show sid
  106 
  107 instance FromHttpApiData Sid where
  108     parseUrlPiece = Right . readText

  110 -- | Synthesis tree
  111 data Tree m tag v x t = Tree
  112     { sID :: Sid
  113     , sState :: SynthesisState m tag v x t
  114     , sDecision :: SynthesisDecision (SynthesisState m tag v x t) m
  115     , sSubForestVar :: TMVar [Tree m tag v x t]
  116     -- ^ lazy mutable field with different synthesis options and sub nodes
  117     }
  118 
  119 targetUnit = mUnit . sTarget . sState
  120 targetDFG = mDataFlowGraph . sTarget . sState
  121 
  122 data SynthesisDecision ctx m where
  123     Root :: SynthesisDecision ctx m
  124     SynthesisDecision ::
  125         (Typeable p, SynthesisDecisionCls ctx m o d p, Show d, ToJSON p, Viewable d DecisionView) =>
  126         {option :: o, decision :: d, metrics :: p, score :: Float} ->
  127         SynthesisDecision ctx m
  128 
  129 class SynthesisDecisionCls ctx m o d p | ctx o -> m d p where
  130     decisions :: ctx -> o -> [(d, m)]
  131     parameters :: ctx -> o -> d -> p
  132     estimate :: ctx -> o -> d -> p -> Float
  133 
  134 data SynthesisState m tag v x t = SynthesisState
  135     { sParent :: Maybe (Tree m tag v x t)
  136     , sTarget :: m
  137     , sAllocationOptions :: [Allocation tag]
  138     -- ^ PU allocation options cache
  139     , sBindOptions :: [Bind tag v x]
  140     -- ^ bind options cache
  141     , sResolveDeadlockOptions :: [ResolveDeadlock v x]
  142     , sOptimizeAccumOptions :: [OptimizeAccum v x]
  143     , sConstantFoldingOptions :: [ConstantFolding v x]
  144     , sBreakLoopOptions :: [BreakLoop v x]
  145     , sDataflowOptions :: [DataflowSt tag v (TimeConstraint t)]
  146     -- ^ dataflow options cache
  147     , bindingAlternative :: M.Map (F v x) [tag]
  148     -- ^ a map from functions to possible processor unit tags
  149     , possibleDeadlockBinds :: S.Set (F v x)
  150     -- ^ a function set, which binding may cause dead lock
  151     , bindWaves :: M.Map v Int
  152     -- ^ if algorithm will be represented as a graph, where nodes -
  153     --  variables of not binded functions, edges - casuality, wave is a
  154     --  minimal number of a step from an initial node to selected
  155     , processWaves :: [ProcessWave v x]
  156     -- ^ Execution waves of the algorithm. See detailed description in NITTA.Intermediate.Analysis module.
  157     , numberOfProcessWaves :: Int
  158     -- ^ Number of execution waves of the algorithm.
  159     , numberOfDataflowOptions :: Int
  160     -- ^ number of dataflow options
  161     , transferableVars :: S.Set v
  162     -- ^ a variable set, which can be transferred on the current
  163     --  synthesis step
  164     }
