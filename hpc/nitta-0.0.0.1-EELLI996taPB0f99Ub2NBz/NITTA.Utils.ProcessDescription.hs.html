   46 -- |Process builder state.
   47 data Schedule pu v x t = Schedule
   48     { -- |Defining process.
   49       schProcess :: Process t (StepInfo v x t)
   50     , -- |Proxy for process unit instruction, which is needed for API simplify. Without that,
   51       -- for some function, the user needs to describe type explicitly.
   52       iProxy :: Proxy (Instruction pu)
   53     }
   54 
   55 instance {-# OVERLAPS #-} NextTick (Schedule pu v x t) t where
   56     nextTick = nextTick . schProcess
   57 
   58 {- |Execute process builder and return new process description. The initial process state is getting
   59 from the PU by the 'process' function.
   60 -}
   61 execSchedule pu st = snd $ runSchedule pu st
   62 
   63 {- |Execute process builder and return new process description. The initial process state is passed
   64 explicetly.
   65 -}
   66 execScheduleWithProcess pu p st = snd $ runScheduleWithProcess pu p st
   67 
   68 {- |Execute process builder and return list of new step UID and new process description. The initial
   69 process state is getting from the PU by the 'process' function.
   70 -}
   71 runSchedule pu st = runScheduleWithProcess pu (process pu) st
   72 
   73 {- |Execute process builder and return list of new step UID and new process description. The initial
   74 process state is passed explicetly.
   75 -}
   76 runScheduleWithProcess pu p st =
   77     let (a, s) =
   78             runState
   79                 st
   80                 Schedule
   81                     { schProcess = p
   82                     , iProxy = ip pu
   83                     }
   84      in (a, schProcess s)
   85     where
   86         ip :: pu -> Proxy (Instruction pu)
   87         ip _ = Proxy
   88 
   89 -- |Add process step with passed the time and info.
   90 scheduleStep placeInTime stepInfo =
   91     scheduleStep' (\uid -> Step uid placeInTime stepInfo)
   92 
   93 scheduleStep' mkStep = do
   94     sch@Schedule{schProcess = p@Process{nextUid, steps}} <- get
   95     put
   96         sch
   97             { schProcess =
   98                 p
   99                     { nextUid = succ nextUid
  100                     , steps = mkStep nextUid : steps
  101                     }
  102             }
  103     return [nextUid]

  105 {- |Add to the process description information about vertical relations, which are defined by the
  106  Cartesian product of high and low lists.
  107 -}
  108 establishVerticalRelations high low = do
  109     sch@Schedule{schProcess = p@Process{relations}} <- get
  110     put
  111         sch
  112             { schProcess =
  113                 p
  114                     { relations = [Vertical h l | h <- high, l <- low] ++ relations
  115                     }
  116             }
  117 
  118 -- |Add to the process description information about vertical relation.
  119 establishVerticalRelation h l = do
  120     sch@Schedule{schProcess = p@Process{relations}} <- get
  121     put
  122         sch
  123             { schProcess =
  124                 p
  125                     { relations = Vertical h l : relations
  126                     }
  127             }
  128 
  129 scheduleFunctionBind f = do
  130     schedule <- get
  131     scheduleStep (singleton $ nextTick schedule) $ CADStep $ "bind " <> show f
  132 
  133 scheduleFunctionRevoke f = do
  134     schedule <- get
  135     scheduleStep (singleton $ nextTick schedule) $ CADStep $ "revoke " <> show f
  136 
  137 -- |Add to the process description information about function evaluation.
  138 scheduleFunction ti f = scheduleStep ti $ FStep f
  139 
  140 {- |Add to the process description information about endpoint behaviour, and it's low-level
  141 implementation (on instruction level). Vertical relations connect endpoint level and instruction
  142 level steps.
  143 -}
  144 scheduleEndpoint EndpointSt{epAt, epRole} codeGen = do
  145     high <- scheduleStep epAt $ EndpointRoleStep epRole
  146     low <- codeGen
  147     establishVerticalRelations high low
  148     return high
  149 
  150 scheduleEndpoint_ ep codeGen = void $ scheduleEndpoint ep codeGen
  151 
  152 {- |Add to the process description information about instruction evaluation.
  153 Unsafe means: without instruction collision check and nextTick consistency.
  154 -}
  155 scheduleInstructionUnsafe at instr = do
  156     Schedule{iProxy} <- get
  157     buf <- scheduleStep at $ InstructionStep (instr `asProxyTypeOf` iProxy)
  158     updateTick $ sup at + 1
  159     return buf
  160     where
  161         updateTick tick = do
  162             sch@Schedule{schProcess} <- get
  163             put
  164                 sch
  165                     { schProcess =
  166                         schProcess
  167                             { nextTick_ = tick
  168                             }
  169                     }
  170 
  171 scheduleInstructionUnsafe_ ti instr = void $ scheduleInstructionUnsafe ti instr
  172 
  173 -- |Add to the process description information about nested step.
  174 scheduleNestedStep tag step@Step{pInterval} = do
  175     pID <- scheduleStep' (\uid -> Step uid pInterval $ NestedStep tag step)
  176     when (length pID /= 1) $ error "scheduleNestedStep internal error."
  177     return $ head pID

  179 -- |Get a current slice of the computational process.
  180 getProcessSlice :: State (Schedule pu v x t) (Process t (StepInfo v x t))
  181 getProcessSlice = do
  182     Schedule{schProcess} <- get
  183     return schProcess
