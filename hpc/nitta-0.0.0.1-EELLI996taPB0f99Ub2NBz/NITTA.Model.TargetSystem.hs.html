   32 {- |Model of target unit, which is a main subject of synthesis process and
   33 synthesis graph.
   34 -}
   35 data TargetSystem u tag v x t = TargetSystem
   36     { -- |model of target unit
   37       mUnit :: u
   38     , -- |whole application algorithm
   39       mDataFlowGraph :: DataFlowGraph v x
   40     }
   41     deriving (Generic)
   42 
   43 instance (WithFunctions u (F v x)) => WithFunctions (TargetSystem u tag v x t) (F v x) where
   44     functions TargetSystem{mUnit, mDataFlowGraph} =
   45         assert (S.fromList (functions mUnit) == S.fromList (functions mDataFlowGraph)) $ -- inconsistent TargetSystem
   46             functions mUnit
   47 
   48 processDuration TargetSystem{mUnit} = nextTick mUnit - 1
   49 
   50 isSynthesisComplete :: (ProcessorUnit u v x t) => TargetSystem u tag v x t -> Bool
   51 isSynthesisComplete TargetSystem{mUnit, mDataFlowGraph} =
   52     transferred mUnit == variables mDataFlowGraph
   53 
   54 instance (BindProblem u tag v x) => BindProblem (TargetSystem u tag v x t) tag v x where
   55     bindOptions TargetSystem{mUnit} = bindOptions mUnit
   56 
   57     bindDecision f@TargetSystem{mUnit} d = f{mUnit = bindDecision mUnit d}
   58 
   59 instance (DataflowProblem u tag v t) => DataflowProblem (TargetSystem u tag v x t) tag v t where
   60     dataflowOptions TargetSystem{mUnit} = dataflowOptions mUnit
   61 
   62     dataflowDecision f@TargetSystem{mUnit} d = f{mUnit = dataflowDecision mUnit d}
   63
