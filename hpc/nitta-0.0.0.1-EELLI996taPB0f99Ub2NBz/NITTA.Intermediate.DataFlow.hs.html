   32 {- |Data flow graph - intermediate representation of application algorithm.
   33 Right now can be replaced by @[F v x]@, but for future features like
   34 conduction statement, we don't do that.
   35 -}
   36 data DataFlowGraph v x
   37     = DFLeaf (F v x)
   38     | DFCluster [DataFlowGraph v x]
   39     deriving (Show, Generic)
   40 
   41 instance Default (DataFlowGraph v x) where
   42     def = DFCluster []
   43 
   44 instance Eq (DataFlowGraph v x) where
   45     -- `show` used for avoid `Ord (DataFlowGraph v x)`
   46     (DFCluster c1) == (DFCluster c2) = S.fromList (map show c1) == S.fromList (map show c2)
   47     (DFLeaf f1) == (DFLeaf f2) = f1 == f2
   48     _ == _ = False
   49 
   50 instance (Var v) => Variables (DataFlowGraph v x) v where
   51     variables (DFLeaf fb) = variables fb
   52     variables (DFCluster g) = unionsMap variables g
   53 
   54 instance WithFunctions (DataFlowGraph v x) (F v x) where
   55     functions (DFLeaf f) = [f]
   56     functions (DFCluster g) = concatMap functions g
   57 
   58 instance (Var v, Val x) => BreakLoopProblem (DataFlowGraph v x) v x where
   59     breakLoopOptions _dfg = []
   60 
   61     breakLoopDecision dfg bl =
   62         let origin = recLoop bl
   63          in fsToDataFlowGraph $
   64                 (recLoopIn bl){funHistory = [origin]} :
   65                 (recLoopOut bl){funHistory = [origin]} :
   66                 (functions dfg L.\\ [origin])
   67 
   68 instance (Var v, Val x) => ConstantFoldingProblem (DataFlowGraph v x) v x where
   69     constantFoldingOptions _dfg = []
   70 
   71     constantFoldingDecision dfg ref@ConstantFolding{} =
   72         fsToDataFlowGraph $ constantFoldingDecision (functions dfg) ref
   73 
   74 instance (Var v, Val x) => OptimizeAccumProblem (DataFlowGraph v x) v x where
   75     optimizeAccumOptions _dfg = []
   76 
   77     optimizeAccumDecision dfg ref@OptimizeAccum{} =
   78         fsToDataFlowGraph $ optimizeAccumDecision (functions dfg) ref
   79 
   80 instance (Var v) => ResolveDeadlockProblem (DataFlowGraph v x) v x where
   81     resolveDeadlockOptions _dfg = []
   82 
   83     resolveDeadlockDecision dfg ResolveDeadlock{newBuffer, changeset} =
   84         fsToDataFlowGraph (newBuffer : map (patch changeset) (functions dfg))
   85 
   86 -- |Convert @[ F v x ]@ to 'DataFlowGraph'.
   87 fsToDataFlowGraph fs = DFCluster $ map DFLeaf fs
   88 
   89 addFuncToDataFlowGraph f (DFCluster items) = DFCluster (DFLeaf f : items)
   90 addFuncToDataFlowGraph f leaf = DFCluster [DFLeaf f, leaf]
