   32 -- |Functional algorithm simulation
   33 simulateDataFlowGraph ::
   34     (Var v, Val x, WithFunctions g (F v x)) =>
   35     Int ->
   36     CycleCntx v x ->
   37     [(v, [x])] ->
   38     g ->
   39     Cntx v x
   40 simulateDataFlowGraph cycleN cycle0 transmission dfg =
   41     simulateAlg cycleN cycle0 transmission $ reorderAlgorithm $ functions dfg
   42 
   43 simulateAlg ::
   44     (Var v, Val x) =>
   45     Int ->
   46     CycleCntx v x ->
   47     [(v, [x])] ->
   48     [F v x] ->
   49     Cntx v x
   50 simulateAlg cycleN cycle0 transmission alg
   51     | let cycleConnections [] = []
   52           cycleConnections (f : fs)
   53             -- without refactoring
   54             | Just (Loop _ (O ov) (I iv)) <- castF f = (iv, elems ov) : cycleConnections fs
   55             -- after refactoring (BreakLoopD)
   56             | Just (LoopBegin (Loop _ (O ov) (I iv)) _) <- castF f = (iv, elems ov) : cycleConnections fs
   57             | otherwise = cycleConnections fs
   58 
   59           fromPrevCycle = cycleConnections alg =
   60         Cntx
   61             { cntxReceived = M.fromList transmission
   62             , cntxProcess = take cycleN $ simulateAlg' fromPrevCycle cycle0 transmission alg
   63             , cntxCycleNumber = cycleN
   64             }
   65 
   66 simulateAlg' fromPrevCycle cycleCntx0 transmission alg =
   67     let (cycleCntx0', transmission') = receive' cycleCntx0 transmission
   68         cycleCntx = simulateCycle cycleCntx0' alg
   69      in cycleCntx : simulateAlg' fromPrevCycle (throwLoop cycleCntx) transmission' alg
   70     where
   71         -- TODO: receive data for several IO processor unit.
   72         receive' CycleCntx{cycleCntx} trans =
   73             ( CycleCntx $
   74                 foldl
   75                     ( \c (v, xs) ->
   76                         case xs of
   77                             x : _ -> HM.insert v x c
   78                             _ -> c
   79                     )
   80                     cycleCntx
   81                     trans
   82             , map
   83                 ( \(v, xs) ->
   84                     case xs of
   85                         (_ : _) -> (v, tail xs)
   86                         [] -> (v, xs)
   87                 )
   88                 trans
   89             )
   90         throwLoop (CycleCntx cntx) =
   91             CycleCntx $
   92                 HM.fromList $
   93                     foldl
   94                         ( \st (thrown, vs) -> map (\v -> (v, cntx HM.! thrown)) vs ++ st
   95                         )
   96                         []
   97                         fromPrevCycle
   98         simulateCycle cntx00 fs =
   99             foldl
  100                 ( \cntx f ->
  101                     case updateCntx cntx $ simulate cntx f of
  102                         Right cntx' -> cntx'
  103                         Left e -> error [i|can't simulate #{f} in context #{cntx}: #{e}|]
  104                 )
  105                 cntx00
  106                 fs
  107 
  108 reorderAlgorithm alg = orderAlgorithm' [] alg
  109     where
  110         orderAlgorithm' _ [] = []
  111         orderAlgorithm' vs fs
  112             | loops@(_ : _) <- filter isLoop fs
  113               , let loopOutputs = elems $ unionsMap outputs loops =
  114                 case filter (not . null . intersect loopOutputs . elems . inputs) loops of
  115                     [] -> loops ++ orderAlgorithm' (elems (unionsMap variables loops) ++ vs) (fs \\ loops)
  116                     ready -> ready ++ orderAlgorithm' (elems (unionsMap variables ready) ++ vs) (fs \\ ready)
  117         orderAlgorithm' vs fs
  118             | let ready = filter (null . (\\ vs) . elems . inputs) fs
  119               , not $ null ready =
  120                 ready ++ orderAlgorithm' (elems (unionsMap variables ready) ++ vs) (fs \\ ready)
  121         orderAlgorithm' _ remain = error $ "Can't sort algorithm: " ++ show remain
