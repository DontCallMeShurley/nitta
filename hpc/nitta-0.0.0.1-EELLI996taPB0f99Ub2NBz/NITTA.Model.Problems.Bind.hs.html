   16 module NITTA.Model.Problems.Bind (
   17     Bind (..),
   18     BindProblem (..),
   19 ) where
   20 
   21 import Data.String.ToString
   22 import GHC.Generics
   23 import NITTA.Intermediate.Types
   24 
   25 data Bind tag v x
   26     = Bind (F v x) tag
   27     deriving (Generic)
   28 
   29 instance (ToString tag) => Show (Bind tag v x) where
   30     show (Bind f tag) = "Bind " <> show f <> " " <> toString tag
   31 
   32 class BindProblem u tag v x | u -> tag v x where
   33     bindOptions :: u -> [Bind tag v x]
   34     bindDecision :: u -> Bind tag v x -> u
   35 
   36 instance (Var v) => Variables (Bind tab v x) v where
   37     variables (Bind f _tag) = variables f
