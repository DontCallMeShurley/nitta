   41 data BuilderSt tag v x t = BuilderSt
   42     { signalBusWidth :: Int
   43     , availSignals :: [SignalTag]
   44     , pus :: M.Map tag (PU v x t)
   45     }
   46 
   47 netIOPorts pus =
   48     BusNetworkIO
   49         { extInputs = setOf puInputPorts
   50         , extOutputs = setOf puOutputPorts
   51         , extInOuts = setOf puInOutPorts
   52         }
   53     where
   54         setOf ioPorts = unionsMap ioPorts pus
   55 
   56 modifyNetwork net@BusNetwork{bnPus, bnSignalBusWidth, bnEnv} builder =
   57     let st0 =
   58             BuilderSt
   59                 { signalBusWidth = bnSignalBusWidth
   60                 , availSignals = map (SignalTag . controlSignalLiteral) [bnSignalBusWidth :: Int ..]
   61                 , pus = bnPus
   62                 }
   63         BuilderSt{signalBusWidth, pus} = execState builder st0
   64      in net
   65             { bnPus = pus
   66             , bnSignalBusWidth = signalBusWidth
   67             , bnEnv = bnEnv{ioPorts = Just $ netIOPorts $ M.elems pus}
   68             }
   69 
   70 defineNetwork bnName ioSync builder = modifyNetwork (busNetwork bnName ioSync) builder
   71 
   72 puEnv tag ctrlPorts ioPorts =
   73     def
   74         { ctrlPorts = Just ctrlPorts
   75         , ioPorts = Just ioPorts
   76         , valueIn = Just ("data_bus", "attr_bus")
   77         , valueOut = Just (toText tag <> "_data_out", toText tag <> "_attr_out")
   78         }
   79 
   80 -- |Add PU with the custom initial state. Type specify by IOPorts.
   81 addCustom tag pu ioPorts = do
   82     st@BuilderSt{signalBusWidth, availSignals, pus} <- get
   83     let ctrlPorts = takePortTags availSignals pu
   84         pu' = PU pu def $ puEnv tag ctrlPorts ioPorts
   85         usedPortsLen = length $ usedPortTags ctrlPorts
   86     put
   87         st
   88             { signalBusWidth = signalBusWidth + usedPortsLen
   89             , availSignals = drop usedPortsLen availSignals
   90             , pus =
   91                 if M.member tag pus
   92                     then error "every PU must has uniq tag"
   93                     else M.insert tag pu' pus
   94             }
   95 
   96 -- |Add PU with the default initial state. Type specify by IOPorts.
   97 add tag ioport = addCustom tag def ioport

   99 data MicroarchitectureDesc tag = MicroarchitectureDesc
  100     { networks :: [NetworkDesc tag]
  101     , ioSyncMode :: IOSynchronization
  102     }
  103     deriving (Generic)
  104 
  105 instance (ToJSON tag) => ToJSON (MicroarchitectureDesc tag)
  106 
  107 data NetworkDesc tag = NetworkDesc
  108     { networkTag :: tag
  109     , valueType :: T.Text
  110     , units :: [UnitDesc tag]
  111     }
  112     deriving (Generic)
  113 
  114 instance (ToJSON tag) => ToJSON (NetworkDesc tag)
  115 
  116 data UnitDesc tag = UnitDesc
  117     { unitTag :: tag
  118     , unitType :: T.Text
  119     }
  120     deriving (Generic)
  121 
  122 instance (ToJSON tag) => ToJSON (UnitDesc tag)
  123 
  124 microarchitectureDesc :: forall tag v x t. (Typeable x) => BusNetwork tag v x t -> MicroarchitectureDesc tag
  125 microarchitectureDesc BusNetwork{bnName, bnPus, ioSync} =
  126     MicroarchitectureDesc
  127         { networks =
  128             [ NetworkDesc
  129                 { networkTag = bnName
  130                 , valueType = showText $ typeRep (Proxy :: Proxy x)
  131                 , units =
  132                     map
  133                         ( \(tag, PU{unit}) ->
  134                             UnitDesc
  135                                 { unitTag = tag
  136                                 , unitType = T.pack $ takeWhile (' ' /=) $ show $ typeOf unit
  137                                 }
  138                         )
  139                         $ M.assocs bnPus
  140                 }
  141             ]
  142         , ioSyncMode = ioSync
  143         }
