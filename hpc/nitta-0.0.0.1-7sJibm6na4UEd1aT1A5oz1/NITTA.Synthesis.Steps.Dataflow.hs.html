   36 data DataflowMetrics = DataflowMetrics
   37     { pWaitTime :: Float
   38     , pRestrictedTime :: Bool
   39     , -- |number of variables, which is not transferable for affected
   40       -- functions.
   41       pNotTransferableInputs :: [Float]
   42     }
   43     deriving (Generic)
   44 
   45 instance ToJSON DataflowMetrics
   46 
   47 instance
   48     (UnitTag tag, VarValTime v x t) =>
   49     SynthesisDecisionCls
   50         (SynthesisState (TargetSystem (BusNetwork tag v x t) tag v x t) tag v x t)
   51         (TargetSystem (BusNetwork tag v x t) tag v x t)
   52         (DataflowSt tag v (TimeConstraint t))
   53         (DataflowSt tag v (Interval t))
   54         DataflowMetrics
   55     where
   56     decisions SynthesisState{sTarget} o = let d = dataflowOption2decision o in [(d, dataflowDecision sTarget d)]
   57 
   58     parameters SynthesisState{transferableVars, sTarget} DataflowSt{dfSource, dfTargets} _ =
   59         let TimeConstraint{tcAvailable, tcDuration} = epAt $ snd dfSource
   60          in DataflowMetrics
   61                 { pWaitTime = fromIntegral (inf tcAvailable)
   62                 , pRestrictedTime = fromEnum (sup tcDuration) /= maxBound
   63                 , pNotTransferableInputs =
   64                     let fs = functions $ mUnit sTarget
   65                         vs = unionsMap (variables . snd) dfTargets
   66                         affectedFunctions = filter (\f -> not $ null (inputs f `S.intersection` vs)) fs
   67                         notTransferableVars = map (\f -> inputs f S.\\ transferableVars) affectedFunctions
   68                      in map (fromIntegral . length) notTransferableVars
   69                 }
   70 
   71     estimate SynthesisState{numberOfDataflowOptions} _o _d DataflowMetrics{pWaitTime, pNotTransferableInputs, pRestrictedTime} =
   72         2000
   73             + (numberOfDataflowOptions >= threshold) <?> 1000
   74             + pRestrictedTime <?> 200
   75             - sum pNotTransferableInputs * 5
   76             - pWaitTime
   77 
   78 threshold = 20
