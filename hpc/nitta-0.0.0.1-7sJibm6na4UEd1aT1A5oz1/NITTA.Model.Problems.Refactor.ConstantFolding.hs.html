   59 import Data.Default
   60 import qualified Data.HashMap.Strict as HM
   61 import qualified Data.List as L
   62 import qualified Data.Set as S
   63 import GHC.Generics
   64 import NITTA.Intermediate.Functions
   65 import NITTA.Intermediate.Types
   66 
   67 data ConstantFolding v x = ConstantFolding
   68     { cRefOld :: [F v x]
   69     , cRefNew :: [F v x]
   70     }
   71     deriving (Generic, Show, Eq)
   72 
   73 class ConstantFoldingProblem u v x | u -> v x where
   74     -- |Function takes algorithm in 'DataFlowGraph' and return list of optimizations that can be done
   75     constantFoldingOptions :: u -> [ConstantFolding v x]
   76     constantFoldingOptions _ = []
   77 
   78     -- |Function takes 'ConstantFolding' and modify 'DataFlowGraph'
   79     constantFoldingDecision :: u -> ConstantFolding v x -> u
   80     constantFoldingDecision _ _ = error "not implemented"
   81 
   82 instance (Var v, Val x) => ConstantFoldingProblem [F v x] v x where
   83     constantFoldingOptions fs =
   84         let clusters = selectClusters fs
   85             evaluatedClusters = map evalCluster clusters
   86             zipOfClusters = zip clusters evaluatedClusters
   87             filteredZip = filter (\case ([_], _) -> False; _ -> True) zipOfClusters
   88             options = [ConstantFolding{cRefOld = c, cRefNew = ec} | (c, ec) <- filteredZip, c /= ec]
   89             optionsFiltered = filter isBlankOptions options
   90             isBlankOptions = not . null . constantFoldingDecision fs
   91          in optionsFiltered
   92 
   93     constantFoldingDecision fs ConstantFolding{cRefOld, cRefNew}
   94         | cRefOld == cRefNew = cRefNew
   95         | otherwise = deleteExtraF $ (fs L.\\ cRefOld) <> cRefNew
   96 
   97 isConst f
   98     | Just Constant{} <- castF f = True
   99     | otherwise = False
  100 
  101 selectClusters fs =
  102     let consts = filter isConst fs
  103         isIntersection a b = not . S.null $ S.intersection a b
  104         inputsAreConst f = inputs f `S.isSubsetOf` S.unions (map outputs consts)
  105         getInputConsts f = filter (\c -> outputs c `isIntersection` inputs f) consts
  106         createCluster f
  107             | inputsAreConst f = f : getInputConsts f
  108             | otherwise = [f]
  109      in map createCluster fs
  110 
  111 evalCluster [f] = [f]
  112 evalCluster fs = outputResult
  113     where
  114         (consts, [f]) = L.partition isConst fs
  115         cntx = CycleCntx $ HM.fromList $ concatMap (simulate def) consts
  116         outputResult
  117             | null $ outputs f = fs
  118             | otherwise = map (\(v, x) -> constant x [v]) (simulate cntx f) <> consts
  119 
  120 deleteExtraF fs =
  121     L.nub
  122         [ f1
  123         | f1 <- fs
  124         , f2 <- fs
  125         , f1 /= f2
  126         , not $ null (variables f1 `S.intersection` variables f2)
  127         ]
