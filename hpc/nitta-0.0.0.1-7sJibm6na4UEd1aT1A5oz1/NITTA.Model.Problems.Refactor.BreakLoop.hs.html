   11 {- |
   12 Module      : NITTA.Model.Problems.Refactor.BreakLoop
   13 Description : Refactoring for support computational loops
   14 Copyright   : (c) Aleksandr Penskoi, 2021
   15 License     : BSD3
   16 Maintainer  : aleksandr.penskoi@gmail.com
   17 Stability   : experimental
   18 
   19 BreakLoop example:
   20 
   21 > BreakLoop x o i
   22 
   23 before:
   24 
   25 > l@( Loop (X x) (O o) (I i) )
   26 
   27 after:
   28 
   29 > LoopEnd l (I i)
   30 > LoopBegin l (O o)
   31 
   32 For more details: 'NITTA.Intermediate.Functions.Loop'
   33 -}
   34 module NITTA.Model.Problems.Refactor.BreakLoop (
   35     BreakLoop (..),
   36     BreakLoopProblem (..),
   37     recLoop,
   38     recLoopIn,
   39     recLoopOut,
   40 ) where
   41 
   42 import qualified Data.Set as S
   43 import GHC.Generics
   44 import NITTA.Intermediate.Functions
   45 import NITTA.Intermediate.Types
   46 
   47 data BreakLoop v x = BreakLoop
   48     { -- |initial looped value
   49       loopX :: x
   50     , -- |output variables
   51       loopO :: S.Set v
   52     , -- |input variable
   53       loopI :: v
   54     }
   55     deriving (Generic, Eq)
   56 
   57 instance (Var v, Val x) => Show (BreakLoop v x) where
   58     show = ("BreakLoop: " <>) . show . recLoop
   59 
   60 class BreakLoopProblem u v x | u -> v x where
   61     breakLoopOptions :: u -> [BreakLoop v x]
   62     breakLoopOptions _ = []
   63 
   64     breakLoopDecision :: u -> BreakLoop v x -> u
   65     breakLoopDecision _ _ = error "not supported"
   66 
   67 recLoop BreakLoop{loopX, loopO, loopI} =
   68     packF $ Loop (X loopX) (O loopO) (I loopI)
   69 
   70 recLoopIn BreakLoop{loopX, loopO, loopI} =
   71     packF $ LoopEnd (Loop (X loopX) (O loopO) (I loopI)) (I loopI)
   72 
   73 recLoopOut BreakLoop{loopX, loopO, loopI} =
   74     packF $ LoopBegin (Loop (X loopX) (O loopO) (I loopI)) (O loopO)
