    |                  |         |              |

    |                  |         |              |
    +------------------+         +--------------+
@

>>> let a = constant 1 ["a"]
>>> let b = constant 2 ["b"]
>>> let res = add "a" "b" ["res"]
>>> loopRes = loop 1 "e" ["res"]
>>> let fs = [a, b, res, loopRes] :: [F String Int]
>>> constantFoldingDecision fs $ head $ constantFoldingOptions fs
